### 自动化测试脚本
```typescript
// 示例: 难度参数验证测试
async function testDifficultyParameters(difficulty: string, expected: DifficultyConfig) {
  await game.setDifficulty(difficulty);
  await game.startNewGame();
  
  const actual = await game.getCurrentDifficultyParams();
  
  assert.equal(actual.fogCoverage, expected.fogCoverage);
  assert.equal(actual.visionRange, expected.visionRange);
  assert.equal(actual.inkMultiplier, expected.inkMultiplier);
}
```
